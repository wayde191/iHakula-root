代码块： 
var contents = [];
var contentsHtml = '';

function initContents(){
  for(var i = 0; i < 100; i++){
    var obj = 'Obj:' + i;
    contents.push(obj);
    contentsHtml += 'tag p' + obj + 'tag p'
  }
}
initContents();

var eleP = document.getElementById('contents');
eleP.innerHTML = contentsHtml;

function filter(elements, filterRuleFunc){
  var res = [];
  for(var i = 0; i < elements.length; i++){
    var ele = elements[i];
    if(filterRuleFunc(ele)){
      res.push(ele);
    }
  }
 return res;
}

var filtedResults = filter(contents, function(object){
  return object.indexOf('1') > -1;
});

var resultHtml = '';
for (var i = 0; i < filtedResults.length; i++){
  resultHtml += 'tag p' + filtedResults[i] + 'tag p';
}
var eleResult = document.getElementById('result');
eleResult.innerHTML = resultHtml;
